create table if not exists
scores
(
    id int not null auto_increment,

    age int check(age between 16 and 24),
    subject varchar(20) not null check(subject in ("数学", "英语", "语文")),

select c1.id, c1.name, c2.ct
    from (
        select * from class_1
            union all
        select * from class_1
    ) as c1 inner join (
        select id, count(id) as ct from (
            select * from class_1
                union all
            select * from class_1
        ) 
        as t 
        group by id
    ) as c2
    on c1.id = c2.id;

/*
还记得这一大坨查询语句吗？

select * from class_1
    union all 
select * from class_1

被重复使用了两次，可不可将
这个查询语句的内容包装成一
个虚拟表呢？

为什么叫做虚拟表？首先确实
含有类似表的数据，所以它是
表，但是它是通过查询语句检
索出来的数据，并不具体存储
，所以叫做虚拟的表。
*/

create view double_c1 as
select * from class_1
    union all
select * from class_1;

select c1.id, c1.name, c2.ct
from 
    double_c1 as c1 
    inner join
    (
        select id, count(id) as ct 
        from double_c1 as t
        group by id
    ) as c2
    on c1.id = c2.id;

-- 视图能更好的重用查询语句，
-- 将查询语句定义成一个视图（变量）的形式查询


drop view double_c1; 
    -- 删除视图

    -- 查看所有视图

    -- 查看所有的存储过程

show procedure status like "avg%";
    -- 查看以 avg 开头的存储过程有哪些

delimiter // -- 设置结尾符为 //

create procedure avg_age()
begin
    select avg(age) as ag
    from class_1;
end //

delimiter ;  -- 设置结尾符为 ;

    -- 创建存储过程（函数），功能是显示 class_1 的平均年龄

drop procedure avg_age;
    -- 删除存储过程

-- -------------------------------------

delimiter //

create procedure fun
(
    out avg_age double,
    out cnt int,
    out old int,
    out young int
)
begin
    select avg(age) 
    into avg_age
    from class_1;

    select count(id) 
    into cnt
    from class_1;

    select max(age) 
    into old
    from class_1;

    select min(age)
    into young
    from class_1;
end //

delimiter ;

call fun
(
    @ag, @cnt, @old, @young
);

select @ag, @cnt, @old, @young;

-- 这个函数可以返回班级 class_1 的平均年龄、人数、最大年龄、最小年龄
-- 并将这些结果定义到变量 ag、cnt、old、young 中
-- 这是 out avg_age int 的作用，输出到变量的意思
-- 还有 in id int 是接受值是传入

# [788. 逆序对的数量](https://www.acwing.com/problem/content/790/)

=== "树状数组，统计区间元素个数"

    ```c++
    #include <bits/stdc++.h>


    const long long N = 2e5;
    long long n;
    long long a[N];
    long long sr[N], len;
    long long tr[N];
    long long res;

    long long lowbit(long long x)
    {
        return x & -x;
    }

    void add(long long id, long long x)
    {
        while (id <= n)
        {
            tr[id] += x;
            id += lowbit(id);
        }
    }

    long long sum(long long id)
    {
        long long ans = 0;
        while (id > 0)
        {
            ans += tr[id];
            id -= lowbit(id);
        }
        return ans;
    }

    long long find(long long x)
    {
        long long l = 1, r = len, mid;
        while (l < r)
        {
            mid = (l + r) >> 1;
            if (sr[mid] < x) l = mid + 1;
            else r = mid;
        }
        
        if (sr[r] == x) return r;
        return -1;
    }

    void solve()
    {
        
        std::cin >> n;
        for (long long i = 1; i <= n; i ++)
        {
            std::cin >> a[i];
            sr[i] = a[i];
        }
        
        std::sort(sr + 1, sr + 1 + n);
        len = std::unique(sr + 1, sr + 1 + n) - sr - 1;
        
        for (long long i = n; i; i --)
        {
            long long key = find(a[i]);
            res += sum(key - 1);
            add(key, 1);
        }
        
        std::cout << res << '\n';
        
        
        
        
    }

    int main()
    {
        std::ios::sync_with_stdio(0);
        std::cin.tie(0); std::cout.tie(0);
        
        
        solve();
        return 0;
    }

    ```

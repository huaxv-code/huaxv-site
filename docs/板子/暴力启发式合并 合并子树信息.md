# [暴力启发式合并 合并子树信息](https://codeforces.com/problemset/problem/600/E)

有一颗树，根节点为 $1$

有 $n$ 个节点，每个节点都有不同的颜色

问每一个节点，以该节点为根的子树中出现最多次的颜色有哪些？

---

考虑暴力启发式合并，按子节点的子树大小来排序子树，将数量少的合并到数量大，将颜色树也合并上去

时间复杂度是：$n \log(n)$

```c++
#include <bits/stdc++.h>

const long long N = 3e6;
long long n;
long long a[N];
long long g[N], e[N], ne[N], ant;

void add(long long x, long long y)
{
    ant ++;
    e[ant] = y;
    ne[ant] = g[x];
    g[x] = ant;
}

long long sm[N], ti[N];
long long size[N];
std::map<long long, long long> mp[N];

void dfs(long long u, long long v)
{
    std::vector<std::pair<long long, long long>> vp;
    
    for (long long i = g[u]; i; i = ne[i])
    {
        if (e[i] == v) continue;
        dfs(e[i], u);
        vp.push_back({e[i], size[e[i]]});
    }
    
    auto cmp = [](const std::pair<long long, long long>& x, const std::pair<long long, long long>& y)
    {
        return x.second < y.second;
    };
    
    std::sort(vp.begin(), vp.end(), cmp);
    long long x = 0, y = 0;
    for (auto [e, s] : vp)
    {
        if (s > size[u])
        {
            x = sm[e];
            y = ti[e];
            mp[u].swap(mp[e]);
        }
        
        for (auto [c, s_] : mp[e])
        {
            long long t = (mp[u][c] += s_);
            if (t > y)
            {
                y = t;
                x = c;
            }
            else if (t == y)
            {
                x += c;
            }
        }
        
        size[u] += s;
        mp[e].clear();
    }
    long long t = (++ mp[u][a[u]]);
    if (t > y)
    {
        y = t; x = a[u];
    }
    else if (t == y) x += a[u];
    size[u] ++;
    sm[u] = x;
    ti[u] = y;
}

void solve()
{
    std::cin >> n;
    for (long long i = 1; i <= n; i ++)
        std::cin >> a[i];
    for (long long i = 1; i < n; i ++)
    {
        long long x, y; std::cin >> x >> y;
        add(x, y); add(y, x);
    }
    
    dfs(1, 0);
    
    for (long long i = 1; i <= n; i ++)
    {
        std::cout << sm[i] << " ";
    }
    std::cout << '\n';
}

int main()
{
    std::ios::sync_with_stdio(0);
    std::cin.tie(0); std::cout.tie(0);
    
    solve();
    return 0;
}

```

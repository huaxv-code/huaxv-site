# [双关键字排序 - 计数排序+基数排序]()

```c++
#include <bits/stdc++.h>

const long long N = 1e6 + 100;
long long n;
long long t1[N], t2[N], id[N], low[N], hig[N];
long long st_[N], cnt[N];

void cnt_sort(long long len) {
    for (long long i = 1; i <= len; i ++) cnt[i] = 0;
    for (long long i = 1; i <= len; i ++) cnt[low[t1[i]]] ++;
    for (long long i = 1; i <= len; i ++) cnt[i] += cnt[i - 1];
    for (long long i = len; i; i --) st_[cnt[low[t1[i]]] --] = i;
    for (long long i = 1; i <= len; i ++) t2[i] = t1[st_[i]];
    
    for (long long i = 1; i <= len; i ++) cnt[i] = 0;
    for (long long i = 1; i <= len; i ++) cnt[hig[t2[i]]] ++;
    for (long long i = 1; i <= len; i ++) cnt[i] += cnt[i - 1];
    for (long long i = len; i; i --) st_[cnt[hig[t2[i]]] --] = i;
    for (long long i = 1; i <= len; i ++) t1[i] = t2[st_[i]];
}

void solve() {
    std::cin >> n;
    for (long long i = 1; i <= n; i ++) {
        t1[i] = id[i] = i;
        std::cin >> hig[i] >> low[i];
    }
    cnt_sort(n);
    for (long long i = 1; i <= n; i ++) {
        std::cout << hig[t1[i]] << ' ' << low[t1[i]] << '\n';
    }
}

int main() {
    std::ios::sync_with_stdio(0);
    std::cin.tie(0);
    std::cout.tie(0);
    
    solve();
    return 0;
}
```

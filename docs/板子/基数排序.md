# [基数排序](https://www.luogu.com.cn/problem/P1177)

先按照个位排序，然后依次拿出来；接着按照 10 位排序，然后依次拿出来；一直类推，依次从低位排序拿出来，然后轮到高位排序拿出来，知道最大位数遍历完为止。

```c++
#include <bits/stdc++.h>

const long long N = 2e5;
long long n;
long long a[N];

long long sn[10][N];
long long ln[N];
// 基数排序
void radix_sort(long long d = 10) {
    // 最大有 10 位，排序 10 轮
    long long cur = 1;
    for (long long _ = 1; _ <= d; _ ++) {
        for (long long i = 1; i <= n; i ++) {
            if (cur > a[i]) {
                sn[0][++ ln[0]] = a[i];
            }
            else {
                long long t = a[i] / cur % 10;
                sn[t][++ ln[t]] = a[i];
            }
        }
        long long p = 0;
        for (long long i = 0; i < 10; i ++) {
            for (long long j = 1; j <= ln[i]; j ++) {
                a[++ p] = sn[i][j];
            }
            ln[i] = 0;
        }
        cur *= 10;
    }
}

void solve() {
    std::cin >> n;
    for (long long i = 1; i <= n; i ++) {
        std::cin >> a[i];
    }
    radix_sort();
    for (long long i = 1; i <= n; i ++) {
        std::cout << a[i] << ' ';
    }
    std::cout << '\n';
}

int main() {
    std::ios::sync_with_stdio(0);
    std::cin.tie(0);
    std::cout.tie(0);
    
    solve();
    return 0;
}
```

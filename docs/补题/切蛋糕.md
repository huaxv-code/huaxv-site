# [切蛋糕]()

假设竖切需要切 k 刀，那么横切就需要 m - k 刀

竖切 k 刀能怎么切？可以观察发现这种状态和二进制数吻合 $1 \sim 2 ^ n$ 种状态，对于每一种状态，二分查找横切 $m - k$ 刀的最大值最小

```c++
#include <bits/stdc++.h>

const long long N = 20;
long long n, m;
long long g[N][N];
long long res;
long long all;

long long lowbit(long long x) {
    return x & -x;
}

long long one_cnt(long long x) {
    long long res = 0;
    while (x) {
        res ++; x -= lowbit(x);
    }
    return res;
}

long long sum[N][N];

long long split_cnt(long long x, long long val) {
    std::vector<long long> d_l = {0};
    long long i = 0;
    long long cur = 1;
    while (cur <= x) {
        if (cur & x) {
            d_l.push_back(i + 1);
        }
        i ++;
        cur <<= 1;
    }
    d_l.push_back(n + 1);
    // return 0;
    long long len = d_l.size();
    long long res = 0;
    std::memset(sum, 0, sizeof sum);
    for (long long i = 1; i < len; i ++) {
        for (long long ln = 1; ln <= n + 1; ln ++) {
            sum[i][ln] = sum[i][ln - 1];
            long long t = 0;
            for (long long j = d_l[i - 1] + 1; j <= d_l[i]; j ++) {
                t += g[ln][j];
            }
            if (t > val) return 1e9;
            sum[i][ln] += t;
        }
    }
    for (long long i = 1; i < len; i ++) {
        long long t = 0;
        long long cnt = 0;
        for (long long j = 1; j <= n + 1; j ++) {
            if (sum[i][j] - sum[i][t] > val) {
                t = j - 1;
                cnt ++;
            }
        }
        res = std::max(res, cnt);
    }
    return res;
}

void solve() {
    std::cin >> n >> m;
    for (long long i = 1; i <= n; i ++) {
        for (long long j = 1; j <= n; j ++) {
            std::cin >> g[i][j];
            all += g[i][j];
        }
    }
    res = all;
    for (long long i = 0; i <= (1 << (n - 1)); i ++) {
        long long x = one_cnt(i);
        if (x > m) break;
        long long y = m - x;
        long long l = 0, r = all, mid;
        // std::cout << "debug: " << x << ' ' << y << '\n';
        while (l < r) {
            mid = (l + r) >> 1;
            // std::cout << mid << " " << split_cnt(i, mid) << '\n';
            if (split_cnt(i, mid) <= y) r = mid;
            else l = mid + 1;
        }
        if (split_cnt(i, r) > y) continue;
        res = std::min(res, r);
    }
    std::cout << res << '\n';
}

int main() {
    std::ios::sync_with_stdio(0);
    std::cin.tie(0);
    std::cout.tie(0);
    
    solve();
    return 0;
}
```

# [切蛋糕]()

假设竖切需要切 k 刀，那么横切就需要 m - k 刀

竖切 k 刀能怎么切？可以观察发现这种状态和二进制数吻合 $1 \sim 2 ^ n$ 种状态，对于每一种状态，二分查找横切 $m - k$ 刀的最大值最小

```c++
#include <bits/stdc++.h>

const long long N = 100;
const long long inf = 1e9;
typedef std::vector<long long> vll;
long long n, m, res = inf;
long long g[N][N];
long long sum[N][N];
long long all;
long long spl[N], len;

void split(long long x) {
    long long cur = 1;
    long long sum = 1;
    len = 0;
    spl[len ++] = 0;
    while (sum <= x) {
        if (sum & x) spl[len ++] = cur;
        sum <<= 1; cur ++;
    }
    if (spl[len - 1] < n) spl[len ++] = n;
}

long long check(long long val) {
    for (long long i = 1; i < len; i ++) {
        for (long long j = 1; j <= n; j ++) {
            sum[i][j] = sum[i][j - 1];
            long long t = 0;
            for (long long k = spl[i - 1] + 1; k <= spl[i]; k ++) {
                t += g[j][k];
            }
            if (t > val) return inf;
            sum[i][j] += t;
        }
    }
    auto get = [&](long long id, long long x, long long y) {
        return sum[id][y] - sum[id][x - 1];
    };
    long long st = 1, ed = 1, cnt = 0;
    while (ed <= n) {
        for (long long i = 1; i < len; i ++) {
            long long t = get(i, st, ed);
            if (st == ed && t > val) return inf;
            if (t > val) {
                st = ed --;
                cnt ++;
                break;
            }
        }
        ed ++;
    }
    return cnt;
}

long long cnt_one(long long x) {
    long long res = 0;
    long long cur = 1;
    while (cur <= x) {
        if (cur & x) res ++;
        cur <<= 1;
    }
    return res;
}

void solve() {
    std::cin >> n >> m;
    for (long long i = 1; i <= n; i ++) {
        for (long long j = 1; j <= n; j ++) {
            std::cin >> g[i][j];
            all += g[i][j];
        }
    }
    long long status = 1 << n; status --;
    for (long long i = 0; i <= status; i ++) {
        long long t = cnt_one(i);
        if (t > m) continue;
        long long i_ = m - t;
        i_ = std::min(i_, n);
        split(i);
        long long l = 0, r = all, mid;
        while (l < r) {
            mid = (l + r) >> 1;
            if (check(mid) <= i_) r = mid;
            else l = mid + 1;
        }
        t = check(r);
        if (t <= i_) {
            res = std::min(res, r);
        }
    }
    std::cout << res << '\n';
}

int main() {
    std::ios::sync_with_stdio(0);
    std::cin.tie(0);
    std::cout.tie(0);
    
    solve();
    return 0;
}
```
